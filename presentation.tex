\begin{frame}{It√¥'s integral (\cite{It√¥1944SI})}
    \heading{Basics}
    \begin{itemize}
        \item  \heading{Definition.} Let \( \Del W_i = W_{t_i} - W_{t_{i-1}} \). For \( X ‚àà L^2_\text{\ad{ad}}\br{[0, 1] √ó Œ©} \) as integrand: take \( L^2(Œ©) \) limits of the \ad{left endpoint evaluation} of Riemann sums
        \[ \ad{M_t} ‚âú ‚à´_0^t \ad{X_s} \dif W_s ‚âú \lim_{n ‚Üí ‚àû} ‚àë_{i = 1}^n \ad{X_{t_{i-1}}} \Del W_i  \quad \text{ in } L^2(Œ©) . \]
        \item  \heading{Example.} \( ‚à´_0^t \ad{W_s} \dif W_s = \frac12 \br{W_t^2 - \alert{t}} \).
    \end{itemize}

    \heading{Properties}
    \begin{itemize}
        \item  Linearity
        \item  Mean: \( 0 \)
        \item  Variance: \( \norm{\ad{M}}_{L^2(Œ©)} = \norm{\ad{X}}_{L^2_\text{\ad{ad}}\br{[0, t] √ó Œ©}} \) (\alert{It√¥'s isometry})
        \item  Martingale: \( \E\br{\ad{M_t} \given ‚Ñ±_s} = \ad{M_s} \) for any \( s ‚â§ t \)
    \end{itemize}
\end{frame}

\begin{frame}{Linear stochastic differential equations (LSDEs)}
    \begin{itemize}
        \item  Linear differential equations incorporating \textquote{noise}, for example
        \[ \frac{\dif \ad{X_t}}{\dif t} = \ad{Œ≤_t} ~ \ad{X_t} + \ad{Œ±_t} ~ \ad{X_t} ~ \alert{\dot{W}_t} . \]

        \item  \alert{But \( \dot{W}_t \) is meaningless.} Heuristically, we multiply by \( \dif t \), write \( \dot{W}_t \dif t = \dif W_t \), and interpret the second expression as an It√¥ integral.

        \item  \heading{Example.} For adapted \( \ad{Œ±} \) and \( \ad{Œ≤} \), the following is an LSDE
        \begin{equation*}
            \left\{
            \begin{aligned}
                \dif \ad{X_t}  & =  \ad{Œ±_t} ~ \ad{X_t} \dif W_t + \ad{Œ≤_t} ~ \ad{X_t} \dif t , \\
                \ad{X_0}  & =  1 .
            \end{aligned}
            \right.
        \end{equation*}

        \item  The solution is given by the \emph{exponential process}
        \[ \ad{‚Ñ∞_t} = \exp\br{‚à´_0^t \ad{Œ±_s} \dif W_s + ‚à´_0^t \br{\ad{Œ≤_s} - \frac12 \ad{Œ±_s^2}} \dif s} . \]
    \end{itemize}
\end{frame}

\begin{frame}{Definition (\cite{AyedKuo2008})}
    \begin{itemize}
        \item  For \ad{\( X \) adapted} and \ii{\( Y \) instantly-independent}, define
        \begin{equation*}
            ‚à´_0^1 \ad{\ad{X_t}} ~ \ii{Y^t} \dif W_t = \lim_{m ‚Üí ‚àû} ‚àë_{j = 1}^m \ad{X_{t_{j - 1}}} \ii{Y^{t_j}} \Del W_i  \quad \text{ in } L^2(Œ©).

        \item  Let \( \gen{Z} \) be a stochastic process such that a sequence \( \br{\gen{Z_n}}_{n = 1}^‚àû \) of stochastic processes each of the form above (or linear combinations thereof) satisfies
        \begin{enumerate}

        \end{enumerate}

        \begin{equation*}
            ‚à´_0^1 \gen{Z(t)} \dif W_t = \lim_{n ‚Üí ‚àû}  ‚à´_0^1 \gen{Z_n(t)} \dif W_t  \quad \text{ in } L^2(Œ©) .

    \end{itemize}

\subsection{Extension of It√¥'s isometry}

\begin{frame}{Identity for a simple case}
    \begin{mytheorem}[{\cite[theorem 3.1]{KuoShresthaSinha2021isometry}}]
        Suppose \( f, œï ‚àà C^1(‚Ñù) \) such that
        \[ f(\ad{W_t}) ~ œï(\ii{W_1 - W_t}), ~ f(\ad{W_t}) ~ œï'(\ii{W_1 - W_t}), ~ f'(\ad{W_t}) ~ œï(\ii{W_1 - W_t}) ‚àà L^2([0, 1] √ó Œ©) . \]
        Then \( \E\bs{ ‚à´_0^1 f(\ad{W_t}) ~ œï(\ii{W_1 - W_t}) \dif W_t } = 0 \), and
        \begin{align*}
            &  \E\bs{ \br{‚à´_0^1 f(\ad{W_t}) ~ œï(\ii{W_1 - W_t}) \dif W_t}^2}  =  ‚à´_0^1 \E\bs{ f(\ad{W_t})^2 ~ œï(\ii{W_1 - W_t})^2 } \dif t  \\
            &  \qquad \qquad +  2 ‚à´_0^1 ‚à´_0^t \E\bs{ f(\ad{W_s}) ~ œï'(\ii{W_1 - W_s}) ~ f'(\ad{W_t}) ~ œï(\ii{W_1 - W_t}) } \dif s \dif t .
        \end{align*}
    \end{mytheorem}

    \pause

    \heading{Remark.} The double integral term can take any real value (\cite[example 3.9]{KuoShresthaSinha2021isometry}).
\end{frame}

\begin{frame}{Proof idea}
    \begin{itemize}
        \item<1->  Write integral as \( L^2(Œ©) \)-limit of sums over partitions of \( [0, 1] \).
        \item<1->  Diagonal: Use quadratic variation of \( W \).
        \item<2->  Off-diagonal: Use the expectation and approximation identities.
        \begin{itemize}

\subsection{Near-martingales}

\begin{frame}{Setup}
    \heading{Motivation}
    \begin{itemize}
        \item  Process defined by It√¥ integrals \( \ad{M_t} = ‚à´_0^t \ad{X_s} \dif W_s \) are martingales.

        \item  Are Ayed‚ÄìKuo integrals martingales?

        \item  \heading{Example.} \( \gen{N(t)} =  ‚à´_0^t \gen{W_1} \dif W_s = W_1 W_t - t \).

            Now, \( \E\br{\gen{N(t)} \given ‚Ñ±_s}  =  W_s^2 - s  ‚â†  W_1 W_s - s  =  \gen{N(s)} \), so \alert{not a martingale}.

            However, \( \E\br{\gen{N(s)} \given ‚Ñ±_s} = W_s^2 - s = \E\br{\gen{N(t)} \given ‚Ñ±_s} \).
    \end{itemize}

    \pause

    \begin{definition}[{\cite[definition 2.1]{HwangKuoSait√¥Zhai2017}}]
        An integrable stochastic process \( \gen{N} \) is called a \emph{near-martingale} if \( \E\br{\gen{N(t)} - \gen{N(s)} \given ‚Ñ±_s} = 0 \) almost surely for every \( s ‚â§ t \).
    \end{definition}

    \pause

    \begin{mytheorem}[{\cite[theorem 2.5]{HwangKuoSait√¥Zhai2017}}]
        A process \( \gen{N} \) is a near-martingale if and only if the conditioned process \( \ad{M} \) given by \( \ad{M_t} = \E\br{\gen{N(t)} \given ‚Ñ±_t} \) is a martingale.
    \end{mytheorem}
\end{frame}
